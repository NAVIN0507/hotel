RewriteEngine On

# Simple redirect for reset-password
RewriteRule ^reset-password$ /reset-password.html [L]
RewriteRule ^reset-password/$ /reset-password.html [L]

# Handle other routes
RewriteCond %{REQUEST_FILENAME} !-f
RewriteCond %{REQUEST_FILENAME} !-d
RewriteRule ^([^/]+)/?$ /$1/index.html [L]

# Fallback
RewriteCond %{REQUEST_FILENAME} !-f
RewriteCond %{REQUEST_FILENAME} !-d
RewriteRule . /index.html [L]
